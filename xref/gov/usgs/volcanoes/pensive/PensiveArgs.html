29  public class PensiveArgs {
30      private static final Logger LOGGER = LoggerFactory.getLogger(PensiveArgs.class);
31  
32      public static final String EXAMPLE_CONFIG_FILENAME = "pensive-example.config";
33      public static final String DEFAULT_CONFIG_FILENAME = "pensive.config";
34  
35      public static final String PROGRAM_NAME = "java -jar gov.usgs.volcanoes.pensive.Pensive";
36      public static final String EXPLANATION = "I am the Pensive server\n";
37      public static final String INPUT_TIME_FORMAT = "yyyyMMddHHmm";
38  
39      private static final Parameter[] PARAMETERS = new Parameter[] {};
40  
41      public final boolean verbose;
42      public final long startTime;
43      public final long endTime;
44      public final String configFileName;
45  
46      public PensiveArgs(String[] commandLineArgs) throws Exception {
47          Arguments args = null;
48          args = new Args(PROGRAM_NAME, EXPLANATION, PARAMETERS);
49          args = new ConfigFileArg(DEFAULT_CONFIG_FILENAME, args);
50          args = new CreateConfigArg(EXAMPLE_CONFIG_FILENAME, args);
51          args = new DateRangeArg(INPUT_TIME_FORMAT, args);
52          args = new VerboseArg(args);
53  
54          JSAPResult jsapResult = null;
55          jsapResult = args.parse(commandLineArgs);
56  
57          verbose = jsapResult.getBoolean("verbose");
58          LOGGER.debug("Setting: verbose={}", verbose);
59  
60          Date startDate = jsapResult.getDate("startTime");
61          if (startDate == null)
62              startTime = Long.MIN_VALUE;
63          else
64              startTime = jsapResult.getDate("startTime").getTime();
65          LOGGER.debug("Setting: startTime={}", startTime);
66  
67          Date endDate = jsapResult.getDate("endTime");
68          if (endDate == null)
69              endTime = Long.MIN_VALUE;
70          else
71              endTime = jsapResult.getDate("endTime").getTime();
72          LOGGER.debug("Setting: endTime={}", endTime);
73  
74          configFileName = jsapResult.getString("config-filename");
75          LOGGER.debug("Setting: config-filename={}", configFileName);
76  
77          if (jsapResult.getBoolean("create-config"))
78              System.exit(1);
79      }
80  }
