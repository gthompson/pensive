3   import java.util.ArrayList;
4   import java.util.Date;
5   import java.util.LinkedList;
6   import java.util.List;
7   import java.util.concurrent.BlockingQueue;
8   import java.util.concurrent.LinkedBlockingQueue;
9   
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import gov.usgs.volcanoes.util.configFile.ConfigFile;
14  import gov.usgs.volcanoes.pensive.PlotJob;
15  import gov.usgs.volcanoes.pensive.WaveSource;
16  import gov.usgs.volcanoes.pensive.plot.SubnetPlotter;
17  
18  /**
19   * Create a pool of connections to a single server and assign plot jobs to
20   * connections as they become available.
21   * 
22   * @author Tom Parker
23   * 
24   *         I waive copyright and related rights in the this work worldwide
25   *         through the CC0 1.0 Universal public domain dedication.
26   *         https://creativecommons.org/publicdomain/zero/1.0/legalcode
27   */
28  public abstract class AbstractPlotScheduler implements Runnable {
29  
30      /** my logger */
31      protected static final Logger LOGGER = LoggerFactory.getLogger("gov.usgs");
32  
33      /** number of concurrent connection to the wave server */
34      public static final int DEFAULT_NUMTHREADS = 5;
35  
36      /** pool of plotters, each with their own wave server connection */
37      private List<WaveSource> waveSources;
38  
39      private List<Thread> threads;
40      
41      /** Queue of plot jobs awaiting an available plotter */
42      protected final BlockingQueue<PlotJob> plotJobs;
43  
44      /** list of subnets that feed from my wave server */
45      protected final List<SubnetPlotter> subnets;
46  
47      /** number of connections to my wave server */
48      private final int numThreads;
49  
50      /** name of this server */
51      public final String name;
52  
53      private Date startDate;
54      
55      private Date endDate;
56      
57      private ConfigFile config;
58      
59      abstract protected void schedulePlots();
60      
61      /**
62       * Class constructor
63       * 
64       * @param name
65       *            name given to this wave server in the config file
66       * @param config
67       *            My configuration stanza
68       */
69      public AbstractPlotScheduler(String name, ConfigFile config) {
70  
71          this.name = name;
72          numThreads = config.getInt("threads", DEFAULT_NUMTHREADS);
73          subnets = new LinkedList<SubnetPlotter>();
74          plotJobs = new LinkedBlockingQueue<PlotJob>();
75          threads = new LinkedList<Thread>();
76          this.config = config;
77      }
78  
79      /**
80       * Start threads
81       */
82      public void startWaveSources() {
83          waveSources = new ArrayList<WaveSource>();
84          for (int i = 0; i < numThreads; i++) {
85              String n = name + "-" + i;
86              WaveSource ws = new WaveSource(n, plotJobs, config);
87              waveSources.add(ws);
88              Thread t = new Thread(ws);
89              t.setName(n);
90              t.start();
91              threads.add(t);
92          }
93      	
94      }
95      /**
96       * 
97       * @param subnet
98       *            The subnet to be added
99       */
100     public void add(SubnetPlotter subnet) {
101         subnets.add(subnet);
102     }
103 
104     /**
105      * 
106      * @return count of subnets I have
107      */
108     public int subnetCount() {
109         return subnets.size();
110     }
111 
112     public void stop() {
113     	for (WaveSource ws : waveSources) {
114     		ws.stop();
115     	}
116     }
117      
118     /**
119      * Schedule the next set of plots. Try to catch all exceptions,
120      * ScheduledExecutorService does the wrong thing with exceptions.
121      */
122     public void run() {
123         try {
124             LOGGER.info("Scheduling plots for " + name);
125             schedulePlots();
126         } catch (Exception e) {
127             LOGGER.error("Caught exception heading for scheduler. " + e.getLocalizedMessage());
128         }
129     }
130 }
