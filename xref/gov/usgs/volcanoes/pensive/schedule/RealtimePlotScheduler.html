7   package gov.usgs.volcanoes.pensive.schedule;
8   
9   import gov.usgs.volcanoes.core.configfile.ConfigFile;
10  import gov.usgs.volcanoes.pensive.PlotJob;
11  import gov.usgs.volcanoes.pensive.plot.SubnetPlotter;
12  
13  /**
14   * A scheduler to queue plots for the most recent time slot.
15   * 
16   * @author Tom Parker
17   *
18   */
19  public class RealtimePlotScheduler extends AbstractPlotScheduler {
20  
21    /**
22     * Class constructor.
23     * 
24     * @param name scheduler name
25     * @param config scheduler config
26     */
27    public RealtimePlotScheduler(final String name, final ConfigFile config) {
28      super(name, config);
29    }
30  
31    /**
32     * Schedule the next plot for each subnet.
33     */
34    @Override
35    protected void schedulePlots() {
36      for (final SubnetPlotter subnet : subnets) {
37        try {
38          LOGGER.info("Scheduling subnet " + subnet.subnetName);
39          plotJobs.put(new PlotJob(subnet));
40        } catch (final InterruptedException e) {
41          LOGGER.info("Interrupted. Unable to schedule " + subnet.subnetName);
42        }
43      }
44    }
45  
46  }
